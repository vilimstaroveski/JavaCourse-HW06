3   import hr.fer.zemris.java.custom.scripting.exec.ObjectMultistack;
4   import hr.fer.zemris.java.custom.scripting.exec.ValueWrapper;

6   public class ObjectMultistackDemo {
7   
8   	public static void main(String[] args) {
9   		ObjectMultistack multistack = new ObjectMultistack();
10  		ValueWrapper year = new ValueWrapper(Integer.valueOf(2000));
11  		multistack.push("year", year);
12  		ValueWrapper price = new ValueWrapper(200.51);
13  		multistack.push("price", price);
14  		System.out.println("Current value for year: "
15  				+ multistack.peek("year").getValue());
16  		System.out.println("Current value for price: "
17  				+ multistack.peek("price").getValue());
18  		multistack.push("year", new ValueWrapper(Integer.valueOf(1900)));
19  		System.out.println("Current value for year: "
20  				+ multistack.peek("year").getValue());
21  		multistack.peek("year").setValue(
22  				((Integer) multistack.peek("year").getValue()).intValue() + 50);
23  		System.out.println("Current value for year: "
24  				+ multistack.peek("year").getValue());//1950
25  		multistack.pop("year");
26  		System.out.println("Current value for year: "//2000
27  				+ multistack.peek("year").getValue());
28  		multistack.peek("year").increment("5");
29  		System.out.println("Current value for year: "//2005
30  				+ multistack.peek("year").getValue());
31  		multistack.peek("year").increment(5);
32  		System.out.println("Current value for year: "//2010
33  				+ multistack.peek("year").getValue());
34  		multistack.peek("year").increment(5.0);
35  		System.out.println("Current value for year: "//2015.0
36  				+ multistack.peek("year").getValue());
37  	}
38  }
